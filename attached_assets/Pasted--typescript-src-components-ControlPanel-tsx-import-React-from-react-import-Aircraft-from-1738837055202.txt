// src/hooks/useFuelCalculations.ts
import { useState } from 'react';
import { Aircraft } from '../types';
import { FuelCalculator, WeightBalanceCalculator } from '../utils';

export const useFuelCalculations = (aircraft: Aircraft) => {
  const [results, setResults] = useState(null);

  const calculate = (params: {
    distance: number;
    payload: number;
    altitude: number;
  }) => {
    const fuelCalc = new FuelCalculator(aircraft);
    const wbCalc = new WeightBalanceCalculator(aircraft);

    // Perform calculations and update results
    const fuelBurn = fuelCalc.calculateFuelBurn(params.distance, params.payload);
    const emissions = fuelCalc.calculateEmissions(fuelBurn);
    
    setResults({
      fuelBurn,
      emissions,
      fuelByPhase: {
        taxi: fuelBurn * 0.05,
        takeoff: fuelBurn * 0.1,
        climb: fuelBurn * 0.2,
        cruise: fuelBurn * 0.55,
        descent: fuelBurn * 0.1
      }
    });
  };

  return { results, calculate };
};
