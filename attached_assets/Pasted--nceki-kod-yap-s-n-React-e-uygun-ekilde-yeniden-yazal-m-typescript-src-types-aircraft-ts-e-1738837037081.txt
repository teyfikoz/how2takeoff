export interface Aircraft {
  id: number;
  name: string; 
  maxTakeoffWeight: number;
  emptyWeight: number;
  maxFuelCapacity: number;
  maxPayload: number;
  baseFuelFlow: number;
  fuelEfficiency: number;
  cgLimits: {
    forward: number;
    aft: number;
  };
}

// src/utils/calculations.ts
export class FuelCalculator {
  private aircraft: Aircraft;

  constructor(aircraft: Aircraft) {
    this.aircraft = aircraft;
  }

  calculateFuelBurn(distance: number, payloadWeight: number): number {
    const baseConsumption = this.aircraft.baseFuelFlow * this.aircraft.fuelEfficiency;
    const loadFactor = payloadWeight / this.aircraft.maxPayload;
    
    return baseConsumption * distance * (1 + loadFactor * 0.1);
  }

  calculateEmissions(fuelBurn: number): {co2: number, nox: number} {
    return {
      co2: fuelBurn * 3.16,
      nox: fuelBurn * 0.014
    };
  }
}

// src/utils/weightBalance.ts
export class WeightBalanceCalculator {
  private aircraft: Aircraft;

  constructor(aircraft: Aircraft) {
    this.aircraft = aircraft;
  }

  calculateCG(payload: number, fuelWeight: number): number {
    const totalWeight = this.aircraft.emptyWeight + payload + fuelWeight;
    // CG calculation logic
    return (totalWeight * 0.4); // Simplified for example
  }

  isWithinLimits(cgPercentMAC: number): boolean {
    return cgPercentMAC >= this.aircraft.cgLimits.forward && 
           cgPercentMAC <= this.aircraft.cgLimits.aft;
  }
}
